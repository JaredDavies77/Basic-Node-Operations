function reverseString(inputString) {
  var split = inputString.split(' '); 
  var reverse = [];
    for (let i=split.length-1; i>=0; i--){
      var word = split[i];
      for (let j=0; j<word.length; j++){
       reverse.unshift(word[j]);
      }
      reverse.unshift(' ');
    }
  reverse.shift();
  console.log(reverse.join(''));
}
